using namespace std;

int ans=INT_MAX;
int n;
vector<vector<int>> d;

void dfs(int v, vector<bool> visited, int dis){
    if(v==0&&visited[v]==true){
        ans=min(ans,dis);
        return;
    }
    visited[v]=true;
    bool tag=false;
    for(int i=0;i<n;++i){
        if(visited[i]==false){
            tag=true;
            dis+=d[v][i];
            dfs(i,visited,dis);
            dis-=d[v][i];
        }
    }
    if(tag==false){
        dis+=d[v][0];
        dfs(0,visited,dis);
    }
    visited[v]=false;
}

int main() {
    cin>>n;
    d.resize(n,vector<int>(n,0));
    for(int i=0;i<n;++i){
        for(int j=0;j<n;++j){
            cin>>d[i][j];
        }
    }
    vector<bool> visited(n, false);
    dfs(0, visited, 0);
    cout<<ans<<endl;
}
